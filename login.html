async function cekLogin() { const email = document.getElementById('email').value; const pass = document.getElementById('pass').value; const btn = document.querySelector('.btn-masuk'); const loader = document.getElementById('loading'); if (!email || !pass) { alert("Email dan Password harus diisi!"); return; } // Tampilkan loading & matikan tombol agar tidak diklik dua kali if(loader) loader.style.display = "block"; btn.disabled = true; btn.innerText = "MEMPROSES..."; const urlServer = "https://script.google.com/macros/s/AKfycbykeGLvmiwTuJk1YPpPuLjAxn1NjaLw3_zcbORgyw0iFh7WraLcmyxhp-JHUA4fvxl0bw/exec "; try { const response = await fetch(`${urlServer}?action=login&email=${encodeURIComponent(email)}&pass=${encodeURIComponent(pass)}`); const result = await response.json(); if (result.status === "Aktif") { alert("Login Berhasil!"); window.location.href = "main_menu.html"; } else if (result.status === "Pending") { alert("Akun belum AKTIF. Silakan cek email aktivasi kamu."); } else if (result.status === "Nonaktif") { alert("Akses dicabut oleh Admin."); } else { alert("Email atau Password salah!"); } } catch (err) { console.error(err); alert("Gagal terhubung ke server.
